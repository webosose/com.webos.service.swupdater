include(FindPkgConfig)

pkg_check_modules(GLIB2 REQUIRED glib-2.0)
include_directories(${GLIB2_INCLUDE_DIRS})
webos_add_compiler_flags(ALL ${GLIB2_CFLAGS_OTHER})

pkg_check_modules(OSTREE REQUIRED ostree-1)
include_directories(${OSTREE_INCLUDE_DIRS})
webos_add_compiler_flags(ALL ${OSTREE_CFLAGS_OTHER})

# Basic
webos_machine_dep()
webos_distro_variant_dep()
set(BIN_NAME libostree-sample)

# Compiling
webos_add_compiler_flags(ALL --std=c++11)
file(GLOB_RECURSE SRC_FILES ${CMAKE_CURRENT_SOURCE_DIR}/*.cpp)
include_directories(${CMAKE_CURRENT_SOURCE_DIR})
add_executable(${BIN_NAME} ${SRC_FILES})

# Linking
webos_add_linker_options(ALL --no-undefined)
set(LIBS ${GLIB2_LDFLAGS}
         ${OSTREE_LDFLAGS}
)
target_link_libraries(${BIN_NAME} ${LIBS})

# Install
webos_build_program(NAME ${BIN_NAME})
